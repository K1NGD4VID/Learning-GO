PACKAGE MAIN

IMPORT (
	"FMT"
	"OS"
)

FUNC MAIN() {
	// 1. CHECK IF AN ARGUMENT WAS PASSED
	IF LEN(OS.ARGS) < 2 {
		FMT.PRINTLN("USAGE: GO RUN MAIN.GO <FILENAME>")
		RETURN
	}

	// 2. READ THE FILENAME (FIRST ARGUMENT AFTER THE PROGRAM NAME)
	FILENAME := OS.ARGS[1]

	// 3. PRINT THE FILENAME BACK TO THE TERMINAL
	FMT.PRINTF("THE FILENAME YOU ENTERED IS: %S\N", FILENAME)
}
